64  int ProgramConfiguration::LoadConfiguration(const char * fileName)
65  {
66  XMLDocument * doc = new XMLDocument();
67  XMLError result = doc->LoadFile(fileName);
68  if (result != XML_NO_ERROR)
69  {
70  Console::Error("Failed to load xml file ", fileName);
71  return EXIT_FAILURE;
72  }
73  XMLElement * xmlElem = doc->RootElement()->FirstChildElement("general");
74  if (!xmlElem)
75  {
76  Console::Error("There isn't an element <general>");
77  return EXIT_FAILURE;
78  }
79  if (ParseGeneral(xmlElem) == EXIT_FAILURE)
80  return EXIT_FAILURE;
81  Console::Log("ParseGeneral done");
82  xmlElem = doc->RootElement()->FirstChildElement("infrastructure");
83  if (!xmlElem)
84  {
85  Console::Error("There isn't an element <infrastructure>");
86  return EXIT_FAILURE;
87  }
88  if (ParseInfrastructure(xmlElem) == EXIT_FAILURE)
89  return EXIT_FAILURE;
90  Console::Log("ParseInfrastructure done");
91  xmlElem = doc->RootElement()->FirstChildElement("setup");
92  if (xmlElem)
93  {
94  if (ParseSetup(xmlElem) == EXIT_FAILURE)
95  return EXIT_FAILURE;
96  }
97  Console::Log("ParseSetup done");
98  xmlElem = doc->RootElement()->FirstChildElement("output");
99  if (xmlElem)
100  {
101  if (ParseOutput(xmlElem) == EXIT_FAILURE)
102  return EXIT_FAILURE;
103  }
104  Console::Log("ParseSetup done");
105  delete doc;
106  return EXIT_SUCCESS;
107  }

118  int ProgramConfiguration::ParseGeneral(XMLElement * general)
119  {
120  XMLElement *xmlElem = general->FirstChildElement("socket");
121  if (!xmlElem)
122  {
123  Console::Error("Missing the socket tag <socket> inside <general>");
124  return EXIT_FAILURE;

134  m_start = 0;
135 
136  xmlElem = general->FirstChildElement("log-file");

138  ParseLog(xmlElem, LOG_FILE);

147  if (xmlElem)
148  {
149  unsigned tmp = xmlElem->UnsignedAttribute("value");
150  if (tmp > 0)

165  int ProgramConfiguration::ParseInfrastructure(XMLElement * infrastructure)
166  {
167  XMLElement* xmlElem = infrastructure->FirstChildElement("rsu");
168  if (!xmlElem)
169  {
170  Console::Error("There must be at least one <rsu> inside <infrastructure>");
171  return EXIT_FAILURE;
172  }
173  while (xmlElem)
174  {
175  if (xmlElem->ToComment())
176  {
177  xmlElem = xmlElem->NextSiblingElement();
178  continue; // skip comment nodes
179  }
180  RsuData rsu;
181  // RSU id
182  rsu.id = xmlElem->IntAttribute("id");
183  NS_ASSERT(rsu.id > 0);
184  // RSU position
185  XMLError result = xmlElem->QueryDoubleAttribute("xpos", &rsu.position.x);
186  if (result != XML_NO_ERROR)
187  {
188  Console::Error("Error parsing attribute xpos of <rsu>");
189  return EXIT_FAILURE;
190  }
191  result = xmlElem->QueryDoubleAttribute("ypos", &rsu.position.y);
192  if (result != XML_NO_ERROR)
193  {
194  Console::Error("Error parsing attribute ypos of <rsu>");
195  return EXIT_FAILURE;
196  }
197  XMLElement* circle = xmlElem->FirstChildElement("cam-area");
198  if (circle)
199  {
200  rsu.cam_area = LoadCircle(circle);
201  }
202  circle = xmlElem->FirstChildElement("car-area");
203  if (circle)
204  {
205  rsu.car_area = LoadCircle(circle);
206  }
207  // RSU directions
208  XMLElement *rsudir = xmlElem->FirstChildElement("direction");
209  while (rsudir)
210  {
211  double dir;
212  bool bVal;
213  int iVal;
214  if (rsudir->ToComment())
215  {
216  rsudir = rsudir->NextSiblingElement("direction");
217  continue; // skip comment nodes
218  }
219  if (rsudir->QueryDoubleAttribute("value", &dir) != XML_NO_ERROR)
220  {
221  Console::Error("Error parsing direction inside <rsu>");
222  return EXIT_FAILURE;
223  }
224  Direction dirStruct;
225  dirStruct.direction = dir;
226  if (rsudir->QueryBoolAttribute("approaching", &bVal) == XML_NO_ERROR)
227  dirStruct.approaching = bVal;
228  if (rsudir->QueryBoolAttribute("leaving", &bVal) == XML_NO_ERROR)
229  dirStruct.leaving = bVal;
230  if (rsudir->QueryIntAttribute("approaching-time", &iVal) == XML_NO_ERROR)
231  if (iVal >= 0 && iVal <= 65535)
232  dirStruct.approachingTime = iVal;
233  if (rsudir->QueryIntAttribute("leaving-time", &iVal) == XML_NO_ERROR)
234  if (iVal >= 0 && iVal <= 65535)
235  dirStruct.leavingTime = iVal;
236  rsu.directions.push_back(dirStruct);
237  // RSU direction lanes
238  XMLElement *lane = rsudir->FirstChildElement("lane");
239  while (lane)
240  {
241  if (lane->ToComment())
242  {
243  lane = lane->NextSiblingElement("lane");
244  continue; // skip comment nodes
245  }
246  TLLane tllane;
247  tllane.dir = dir;
248  const char * cstr = lane->Attribute("friendly-name");
249  if (cstr != NULL)
250  {
251  tllane.friendlyName = std::string(cstr);
252  }
253  cstr = lane->Attribute("controlled-lane");
254  if (cstr != NULL)
255  {
256  tllane.controlledLane = std::string(cstr);
257  }
258  cstr = lane->Attribute("following-lane");
259  if (cstr != NULL)
260  {
261  tllane.followingLane = std::string(cstr);
262  }
263 
264  rsu.lanes.push_back(tllane);
265 
266  lane = lane->NextSiblingElement("lane");
267  }
268  rsudir = rsudir->NextSiblingElement("direction");
269  }
270 
271  m_rsus.insert(make_pair(rsu.id, rsu));
272  xmlElem = xmlElem->NextSiblingElement("rsu");
273  }
274  return EXIT_SUCCESS;
275  }

409  int ProgramConfiguration::ParseNodeSampler(tinyxml2::XMLElement * nodeSampler)
410  {
411  using namespace application;
412  int iVal;
413  double dVal;
414  if (nodeSampler->QueryIntAttribute("quantity", &iVal) == XML_NO_ERROR)
415  if (iVal >= 0 && iVal <= 65535)
416  NodeSampler::Quantity = iVal;
417  if (nodeSampler->QueryDoubleAttribute("movement-threshold", &dVal) == XML_NO_ERROR)
418  NodeSampler::MovementThreshold = dVal;
419  if (nodeSampler->QueryDoubleAttribute("steer-filter-threshold", &dVal) == XML_NO_ERROR)
420  NodeSampler::SteerFilterThreshold = dVal;
421  if (nodeSampler->QueryIntAttribute("steer-filter-count", &iVal) == XML_NO_ERROR)
422  if (iVal >= 0 && iVal <= 255)
423  NodeSampler::SteerFilterCount = iVal;
424  if (nodeSampler->QueryDoubleAttribute("speed-bound", &dVal) == XML_NO_ERROR)
425  NodeSampler::SpeedBound = dVal;
426  if (nodeSampler->QueryDoubleAttribute("speed-variance", &dVal) == XML_NO_ERROR)
427  NodeSampler::SpeedVariance = dVal;
428  //Default attributes
429  if (nodeSampler->QueryIntAttribute("default-resolution", &iVal) == XML_NO_ERROR)
430  if (iVal >= 0 && iVal <= 65535)
431  NodeSampler::DefaultResolution = iVal;
432  if (nodeSampler->QueryDoubleAttribute("default-position-radius", &dVal) == XML_NO_ERROR)
433  NodeSampler::DefaultPositionRadius = dVal;
434  if (nodeSampler->QueryDoubleAttribute("default-position-variance", &dVal) == XML_NO_ERROR)
435  NodeSampler::DefaultPositionVariance = dVal;
436  if (nodeSampler->QueryDoubleAttribute("default-direction-variance", &dVal) == XML_NO_ERROR)
437  NodeSampler::DefaultDirectionVariance = dVal;
438  if (nodeSampler->QueryDoubleAttribute("default-speed-error", &dVal) == XML_NO_ERROR)
439  NodeSampler::DefaultSpeedError = dVal;
440  XMLElement *nodeClass = nodeSampler->FirstChildElement("node-class");
441  while (nodeClass)
442  {
443  const char * type = nodeClass->Attribute("type");
444  if (type == NULL)
445  {
446  Console::Warning("No type attribute in a node-class tag. It will be skipped.");
447  continue;
448  }
449  NodeTypeSamplerAttributes classAttribute;
450  if (!NodeSampler::GetNodeTypeFromString(type, classAttribute.nodeType))
451  {
452  ostringstream oss;
453  oss << "Type " << type << " is unknown. It will be skipped.";
454  Console::Warning(oss.str());
455  continue;
456  }
457  if (nodeClass->QueryIntAttribute("resolution", &iVal) == XML_NO_ERROR)
458  if (iVal >= 0 && iVal <= 65535)
459  classAttribute.resolution = iVal;
460  if (nodeClass->QueryDoubleAttribute("position-radius", &dVal) == XML_NO_ERROR)
461  classAttribute.positionRadius = dVal;
462  if (nodeClass->QueryDoubleAttribute("position-variance", &dVal) == XML_NO_ERROR)
463  classAttribute.positionVariance = dVal;
464  if (nodeClass->QueryDoubleAttribute("direction-variance", &dVal) == XML_NO_ERROR)
465  classAttribute.directionVariance = dVal;
466  if (nodeClass->QueryDoubleAttribute("speed-error", &dVal) == XML_NO_ERROR)
467  classAttribute.speedError = dVal;
468  NodeSampler::AddNodeTypeSamplerAttributes(classAttribute);
469  nodeClass = nodeClass->NextSiblingElement("node-class");
470  }
471  return EXIT_SUCCESS;
472  }
