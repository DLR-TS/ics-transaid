65  bool CentralizedProtocol::CompluteSpeed(const NodeInfo & latest, const NodeInfo & first, double & speedAll,

67  {

78  void CentralizedProtocol::AggregateDataForDirection(const NodeDataMap& data, ValueMap& valueMap)
79  {
80  double speed = 0;
81  double spaceMeanSpeed = 0;
82  double speedTime = 0;
83  double speedTimeFilter = 0;
84  double speedTimeSingle = 0;
85  int number = 0;
86  int numberSpeedFilter = 0;
87  double maxd = 0;
88  double mind = 1000 * 1000;
89  double time = 0;
90  int numLastMex = 0;
91  std::ostringstream oss;
92  for (NodeDataMap::const_iterator node = data.begin(); node != data.end(); ++node)
93  {
94  //The latest message is in the last position of the set
95  NodeInfo info = *(node->second.rbegin().operator *());
96  //Do not count the nodes which have sent a no longer conformant message
97  if (!info.toRemove)
98  {
99  ++number;
100  speed += info.currentSpeed;
101  //Space Mean Speed
102  spaceMeanSpeed += 1 / info.currentSpeed;
103  oss << info.currentSpeed << " ";
104  //Speed from messages
105  if (node->second.size() > 1)
106  {
107  double spd, spdF;
108  NodeInfo first = *(node->second.begin().operator *());
109  if (CompluteSpeed(info, first, spd, spdF))
110  {
111  ++numberSpeedFilter;
112  speedTimeFilter += spdF;
113  }
114  speedTime += spd;
115  NodeDataCollection::reverse_iterator rit = node->second.rbegin();
116  ++rit;
117  NodeInfo last = *(rit.operator *());
118  CompluteSpeed(info, last, spd, spdF);
119  speedTimeSingle += spd;
120  }
121  if (info.distance > maxd)
122  maxd = info.distance;
123  if (info.distance < mind)
124  mind = info.distance;
125  }
126  //Use also the data from no longer conformant nodes
127  if (info.lastMessage && info.totalTime > 0)
128  {
129  ++numLastMex;
130  time += info.totalTime;
131  }
132  }
133  if (numLastMex > 0)
134  time /= numLastMex;
135  else
136  time = -1;
137  if (number > 0)
138  {
139  speed /= number;
140  //Space Mean Speed
141  spaceMeanSpeed = number / spaceMeanSpeed;
142  speedTime /= number;
143  speedTimeSingle /= number;
144  std::string tmp = oss.str();
145  NS_LOG_INFO("SpaceMeanSpeed " << spaceMeanSpeed << " n="<<number<<" list=" << tmp);
146  } else
147  {
148  mind = 0;
149  speed = -1;
150  spaceMeanSpeed = -1;
151  speedTime = -1;
152  speedTimeSingle = -1;
153  }
154  if (numberSpeedFilter > 0)
155  speedTimeFilter /= number;
156  else
157  speedTimeFilter = -1;
158  valueMap[NUMBER] = number;
159  valueMap[SPEED] = speed;
160  valueMap[TIME] = time;
161  valueMap[MAX_DISTANCE] = maxd;
162  valueMap[MIN_DISTANCE] = mind;
163  valueMap[SPACE_MEAN_SPEED] = spaceMeanSpeed;
164  valueMap[SPEED_TIME] = speedTime;
165  valueMap[SPEED_TIME_FILTER] = speedTimeFilter;
166  valueMap[SPEED_TIME_SINGLE] = speedTimeSingle;
167 
168  }

170  bool CentralizedProtocol::Execute(DirectionValueMap &data, const DataMap &dataMap)
171  {
172  NS_LOG_FUNCTION("");
173  std::vector<std::string> flows;
174  std::stringstream str;
175  int totNum = 0;
176  for (DataMap::const_iterator dir = dataMap.begin(); dir != dataMap.end(); ++dir)
177  {
178  ValueMap valueMap;
179  AggregateDataForDirection(dir->second, valueMap);
180  totNum += valueMap[NUMBER];
181  std::ostringstream out;
182  out << dir->first << "=" << valueMap[NUMBER] << ":" << valueMap[MAX_DISTANCE] << ":" << valueMap[SPEED] << ":"
183 // << valueMap[TIME] << ":" << valueMap[MIN_DISTANCE] << ":" << valueMap[SPACE_MEAN_SPEED] << ":"
184  << valueMap[TIME] << ":" << valueMap[MIN_DISTANCE] << ":" << valueMap[SPEED_TIME_SINGLE] << ":"
185  << valueMap[SPEED_TIME] << ":" << valueMap[SPEED_TIME_FILTER];
186  flows.push_back(out.str());
187  if (valueMap[SPEED] == -1)
188  valueMap[SPEED] = 0;
189  if (valueMap[TIME] == -1)
190  valueMap[TIME] = 0;
191  data[dir->first] = valueMap;
192  str << "dir=" << dir->first << " num=" << valueMap[NUMBER] << " spd=" << valueMap[SPEED] << " sms="
193 // << valueMap[SPACE_MEAN_SPEED] << " st=" << valueMap[SPEED_TIME] << ":" << valueMap[SPEED_TIME_FILTER]
194  << valueMap[SPEED_TIME_SINGLE] << " st=" << valueMap[SPEED_TIME] << ":" << valueMap[SPEED_TIME_FILTER]
195  << " ";
196  }
197  std::ostringstream out;
198  out << "totc=" << totNum;
199  flows.insert(flows.begin(), out.str());
200  NS_LOG_INFO("CentralizedProtocol " << str.str());
201  m_traceFlows(flows);
202  return ReturnData;
203  }
