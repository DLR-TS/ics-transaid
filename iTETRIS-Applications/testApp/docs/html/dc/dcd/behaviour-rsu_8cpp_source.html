84  void BehaviourRsu::AddDirections(std::vector<Direction> directions)
85  {
86  NS_LOG_FUNCTION(Log());
87  bool wasRunning = false;
88  if (IsRunning())
89  {
90  wasRunning = true;
91  Stop();
92  }
93  if (directions.size() == 0)
94  NS_FATAL_ERROR(Log()<<"the RSU can't have 0 directions");
95  m_directions.clear();
96  for (std::vector<Direction>::const_iterator it = directions.begin(); it != directions.end(); ++it)
97  {
98  if (!it->approaching && !it->leaving)
99  {
100  NS_LOG_WARN(Log() << "direction "<< it->direction << " approaching=false and leaving=false");
101  continue;
102  }
103  if (it->approaching)
104  {
105  VehicleDirection dir(it->direction, APPROACHING, it->approachingTime);
106  m_directions.push_back(dir);
107  }
108  if (it->leaving)
109  {
110  VehicleDirection dir(it->direction, LEAVING, it->leavingTime);
111  m_directions.push_back(dir);
112  }
113  }
114 
115  if (wasRunning)
116  {
117  Start();
118  }
119  }

121  void BehaviourRsu::Start()
122  {
123  if (!m_enabled)
124  return;
125  if (m_directions.size() == 0)
126  NS_FATAL_ERROR(Log()<<"Can't start the RSU with 0 directions");
127  int totTime = 0;
128  for (std::vector<VehicleDirection>::iterator it = m_directions.begin(); it != m_directions.end(); ++it)
129  {
130  if (it->time == 0)
131  {
132  it->time = m_timeBeaconMin;
133  NS_LOG_INFO(Log() << "Direction " << *it << " beacon time was Zero. Set to " << m_timeBeaconMin);
134  } else
135  NS_LOG_INFO(Log() << "Direction " << *it << " beacon time is " << it->time);
136  totTime += it->time;
137  }
138  NS_LOG_INFO(Log() << "Total beacon time will be " << totTime);
139  m_eventBeacon = Scheduler::Schedule(0, &BehaviourRsu::EventBeacon, this, 0);
140  m_eventCheck = Scheduler::Schedule(m_timeCheck, &BehaviourRsu::EventCheck, this);
141  Behaviour::Start();
142  }

161  void BehaviourRsu::Receive(server::Payload *payload, double snr)
162  {
163  NS_LOG_FUNCTION(Log());
164  if (!m_enabled)
165  return;
166  CommHeader* commHeader;
167  GetController()->GetHeader(payload, server::PAYLOAD_FRONT, commHeader);
168  switch (commHeader->getMessageType())
169  {
170  case MT_BEACON_RESPONSE:
171  BeaconResponseHeader* resposeHeader;
172  GetController()->GetHeader(payload, server::PAYLOAD_END, resposeHeader);
173  OnBeaconResponse(commHeader, resposeHeader);
174  break;
175  case MT_NO_LONGHER_CONFORMANT:
176  NoLongerConformantHeader * header;
177  GetController()->GetHeader(payload, server::PAYLOAD_END, header);
178  OnNoLongerConformant(commHeader, header);
179  break;
180  default:
181  NS_LOG_WARN(Log() << "Received unknown message type: "<<commHeader->getMessageType());
182  }
183  }

202  void BehaviourRsu::OnBeaconResponse(CommHeader * commHeader, BeaconResponseHeader * resposeHeader)
203  {
204  NS_LOG_FUNCTION(Log());
205  NodeInfo * info = new NodeInfo();
206  info->nodeId = commHeader->getSourceId();
207  info->position = commHeader->getSourcePosition();
208  info->distance = GetDistance(GetController()->GetPosition(), info->position);
209  info->direction = resposeHeader->getSourceDirection();
210  info->conformantDirection.dir = resposeHeader->getConformantDirection();
211  info->conformantDirection.vMov = resposeHeader->getVehicleMovement();
212  info->currentSpeed = resposeHeader->getCurrentSpeed();
213  info->avgSpeedSmall = resposeHeader->getAvgSpeedLow();
214  info->avgSpeedHigh = resposeHeader->getAvgSpeedHigh();
215  info->lastSeen = CurrentTime::Now();
216  info->lastMessage = resposeHeader->getLastMessage();
217 
218  NS_LOG_INFO(
219  Log() << "received info for node " << info->nodeId << ": pos=" << info->position<< " " << PrintHeader(resposeHeader));
220  if (info->lastMessage)
221  {
222  RemoveLastSeen(info);
223  m_traceLastMessageNode(info);
224  } else
225  {
226  UpdateLastSeen(info);
227  m_traceBeaconResponse(info);
228  }

270  void BehaviourRsu::CheckTimeout()
271  {
272  int currentTime = CurrentTime::Now();
273  for (TimeoutMap::iterator node = m_nodeLastSeen.begin(); node != m_nodeLastSeen.end();)
274  {
275  for (DirMap::iterator dir = node->second.begin(); dir != node->second.end();)
276  {
277  if (currentTime - dir->second > m_timeOut)
278  {
279  NS_LOG_INFO(
280  Log()<<"Node " << node->first << " timeout in direction " << dir->first.getId()<< ". Was last seen " << dir->second);
281  NodeInfo * info = new NodeInfo();
282  info->nodeId = node->first;
283  info->conformantDirection = dir->first;
284  info->lastSeen = dir->second;
285  m_traceTimeOutNode(info);
286  node->second.erase(dir++);
287  } else
288  ++dir;
289  }
290  if (node->second.size() == 0)
291  m_nodeLastSeen.erase(node++);
292  else
293  ++node;
294  }
295  }
