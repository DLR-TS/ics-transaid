138  void DataManager::Start()
139  {
140  if (!m_enabled)
141  return;
142  NS_LOG_FUNCTION(Log());
143  BehaviourRsu* rsu = (BehaviourRsu*) GetController()->GetBehaviour(BehaviourRsu::Type());
144  rsu->TraceConnect("NodeReceiveData", MakeCallback(&DataManager::OnBeaconResponse, this));
145  rsu->TraceConnect("NodeTimeOut", MakeCallback(&DataManager::OnTimeOutNode, this));
146  rsu->TraceConnect("NodeLastMessage", MakeCallback(&DataManager::OnLastMessageNode, this));
147  rsu->TraceConnect("NodeNoLongerConforman", MakeCallback(&DataManager::OnNoLongerConforman, this));
148  m_eventExecute = Scheduler::Schedule(ExecuteTime, &DataManager::EventExecute, this);
149  const std::vector<VehicleDirection> dirs = rsu->GetDirections();
150  for (std::vector<VehicleDirection>::const_iterator dir = dirs.begin(); dir != dirs.end(); ++dir)
151  m_dataMap[dir->getId()] = NodeDataMap();
152  Behaviour::Start();
153  m_executeAtThisStep = false;
154  }

183  bool DataManager::Execute(DirectionValueMap &data)
184  {
185  if (m_executeAtThisStep)
186  {
187  m_executeAtThisStep = false;
188  if (m_dataMap.size() == 0)
189  return false;
190  NS_LOG_FUNCTION(Log() << "Executing protocols");
191 
192  bool executed = false;
193  for (ProtocolList::iterator protocol = m_protocols.begin(); protocol != m_protocols.end(); ++protocol)
194  {
195  ExecuteBase* eb = *(protocol);
196  if (eb->Execute(data, m_dataMap))
197  executed = true;
198  }
199  //Now remove the marked nodes
200  for (RemoveList::iterator it = m_ToRemove.begin(); it != m_ToRemove.end(); ++it)
201  RemoveData(it->first, it->second);
202  m_ToRemove.clear();
203  return executed;
204  }
205  return false;
206  }

213  void DataManager::OnNoLongerConforman(NodeInfo* info)
214  {
215  NodeDataCollection collection;
216  if (GetNodeCollection(info, collection))
217  {
218  NodeInfo* firstMessage = *(collection.begin());
219  NodeInfo* lastMessage = *(collection.rbegin());
220  //Use the time of the last valid message or of the current invalid one?
221  lastMessage->totalTime = lastMessage->lastSeen - firstMessage->lastSeen;
222  lastMessage->toRemove = true;
223  lastMessage->lastMessage = true;
224 // Mark the data for removal on the next execution
225  m_ToRemove.push_back(std::make_pair(info->conformantDirection.getId(), info->nodeId));
226  }
227  delete info;
228  }

234  void DataManager::OnLastMessageNode(NodeInfo* info)
235  {
236  NodeDataCollection collection;
237  if (GetNodeCollection(info, collection))
238  {
239  NodeInfo* firstMessage = *(collection.begin());
240  info->totalTime = info->lastSeen - firstMessage->lastSeen;
241  } else
242  info->totalTime = 0;
243  bool newData = AddData(info);
244 // Mark the data for removal on the next execution
245  m_ToRemove.push_back(std::make_pair(info->conformantDirection.getId(), info->nodeId));
246  NS_LOG_FUNCTION((newData ? "true ":"false ")<<info->nodeId<<" "<<info->lastSeen);
247  }
