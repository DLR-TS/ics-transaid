910 double NormalVariableImpl::GetValue()
911 {
912 
913  RngStream *generator = GetStream();
914  if (m_nextValid)
915  { // use previously generated
916  m_nextValid = false;
917  return m_next;
918  }
919  while (1)
920  { // See Simulation Modeling and Analysis p. 466 (Averill Law)
921  // for algorithm; basically a Box-Muller transform:
922  // http://en.wikipedia.org/wiki/Box-Muller_transform
923  double u1 = generator->RandU01();
924  double u2 = generator->RandU01();
925  double v1 = 2 * u1 - 1;
926  double v2 = 2 * u2 - 1;
927  double w = v1 * v1 + v2 * v2;
928  if (w <= 1.0)
929  { // Got good pair
930  double y = std::sqrt((-2 * std::log(w)) / w);
931  m_next = m_mean + v2 * y * std::sqrt(m_variance);
932  // if next is in bounds, it is valid
933  m_nextValid = std::fabs(m_next - m_mean) <= m_bound;
934  double x1 = m_mean + v1 * y * std::sqrt(m_variance);
935  // if x1 is in bounds, return it
936  if (std::fabs(x1 - m_mean) <= m_bound)
937  {
938  return x1;
939  }
940  // otherwise try and return m_next if it is valid
941  else if (m_nextValid)
942  {
943  m_nextValid = false;
944  return m_next;
945  }
946  // otherwise, just run this loop again
947  }
948  }
949 }

1069 double EmpiricalVariableImpl::GetValue()
1070 { // Return a value from the empirical distribution
1071 // This code based (loosely) on code by Bruce Mah (Thanks Bruce!)
1072 
1073  RngStream *generator = GetStream();
1074  if (emp.size() == 0)
1075  {
1076  return 0.0; // HuH? No empirical data
1077  }
1078  if (!validated)
1079  {
1080  Validate(); // Insure in non-decreasing
1081  }
1082  double r = generator->RandU01();
1083  if (r <= emp.front().cdf)
1084  {
1085  return emp.front().value; // Less than first
1086  }
1087  if (r >= emp.back().cdf)
1088  {
1089  return emp.back().value; // Greater than last
1090  }
1091  // Binary search
1092  std::vector<ValueCDF>::size_type bottom = 0;
1093  std::vector<ValueCDF>::size_type top = emp.size() - 1;
1094  while (1)
1095  {
1096  std::vector<ValueCDF>::size_type c = (top + bottom) / 2;
1097  if (r >= emp[c].cdf && r < emp[c + 1].cdf)
1098  { // Found it
1099  return Interpolate(emp[c].cdf, emp[c + 1].cdf, emp[c].value, emp[c + 1].value, r);
1100  }
1101  // Not here, adjust bounds
1102  if (r < emp[c].cdf)
1103  {
1104  top = c - 1;
1105  }
1106  else
1107  {
1108  bottom = c + 1;
1109  }
1110  }
1111 }

1424  tools/ranvar.cc
1425 
1426  Originally the algorithm was devised by Marsaglia in 2000:
1427  G. Marsaglia, W. W. Tsang: A simple method for gereating Gamma variables
1428  ACM Transactions on mathematical software, Vol. 26, No. 3, Sept. 2000
1429 
1430  The Gamma distribution density function has the form
1431 
1432  x^(alpha-1) * exp(-x/beta)
1433  p(x; alpha, beta) = ----------------------------
1434  beta^alpha * Gamma(alpha)
1435 
1436  for x > 0.
1437  */
1438 double GammaVariableImpl::GetValue(double alpha, double beta)
1439 {
1440 
1441  RngStream *generator = GetStream();
1442 
1443  if (alpha < 1)
1444  {
1445  double u = generator->RandU01();
1446  return GetValue(1.0 + alpha, beta) * std::pow(u, 1.0 / alpha);
1447  }
1448 
1449  double x, v, u;
1450  double d = alpha - 1.0 / 3.0;
1451  double c = (1.0 / 3.0) / std::sqrt(d);
1452 
1453  while (1)
1454  {
1455  do
1456  {
1457  x = m_normal.GetValue();
1458  v = 1.0 + c * x;
1459  } while (v <= 0);
1460 
1461  v = v * v * v;
1462  u = generator->RandU01();
1463  if (u < 1 - 0.0331 * x * x * x * x)
1464  {
1465  break;
1466  }
1467  if (std::log(u) < 0.5 * x * x + d * (1 - v + std::log(v)))
1468  {
1469  break;
1470  }
1471  }
1472 
1473  return beta * d * v;
1474 }
