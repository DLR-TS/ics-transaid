107  void BehaviourNodeWithSink::Receive(server::Payload *payload, double snr)
108  {
109  NS_LOG_FUNCTION(Log());
110  if (!m_enabled)
111  return;
112  CommHeader* commHeader;
113  GetController()->GetHeader(payload, server::PAYLOAD_FRONT, commHeader);
114  if (commHeader->getMessageType() != MT_RSU_BEACON)
115  {
116  NS_LOG_WARN(Log()<< "Received an unknown message "<< commHeader->getMessageType());
117  return;
118  }
119  BeaconHeader* beaconHeader;
120  GetController()->GetHeader(payload, server::PAYLOAD_END, beaconHeader);
121  int rsuId = commHeader->getSourceId();
122  VehicleDirection dir(beaconHeader->getDirection(), beaconHeader->getVehicleMovement());
123  if (m_muteRsu.id == rsuId && m_muteRsu.muted && m_muteRsu.dir == dir)
124  {
125  NS_LOG_INFO(
126  Log() << "The sink threshold of " << m_sinkThreshold << " has been reached for the RSU " << rsuId << " direction " << dir);
127  return;
128  }
129  // if (GetController()->IsConformantDirection(beaconHeader->getDirection()))
130  if (GetController()->IsConformantDirectionAndMovement(dir, commHeader->getSourcePosition()))
131  {
132  NodeInfo rsu;
133  rsu.nodeId = rsuId;
134  rsu.position = commHeader->getSourcePosition();
135  rsu.conformantDirection = dir;
136  if (m_muteRsu.id != rsuId && m_muteRsu.dir != dir)
137  m_muteRsu = RSU(rsuId, dir);
138 
139  double nextTime = m_rnd.GetValue(m_responseTimeSpacing,
140  beaconHeader->getMaxResponseTime() - m_responseTimeSpacing);
141  Scheduler::Cancel(m_eventResponse);
142  m_eventResponse = Scheduler::Schedule(nextTime, &BehaviourNodeWithSink::EventSendResponse, this, rsu);
143  NS_LOG_INFO(Log() << "scheduled a beacon response in " << nextTime);
144  } else
145  {
146  NS_LOG_INFO(
147  Log() << "beacon direction " << dir << " is not conformant with the current direction " << GetController()->GetDirection());
148  }
149  }

151  void BehaviourNodeWithSink::EventSendResponse(NodeInfo rsu)
152  {
153  NS_LOG_FUNCTION(Log());
154  // if (!GetController()->IsConformantDirection(rsu.direction))
155  if (!GetController()->IsConformantDirectionAndMovement(rsu.conformantDirection, rsu.position))
156  {
157  NS_LOG_WARN(
158  Log() << "The node has a direction no longer conformant. CurrDir=" << GetController()->GetDirection() << ", RsuDir=" << rsu.conformantDirection);
159  return;
160  }
161  double distance;
162  bool last = false;
163  if ((distance = GetDistance(rsu.position, GetController()->GetPosition())) < m_sinkThreshold)
164  {
165  if (rsu.conformantDirection.vMov == APPROACHING)
166  {
167  NS_LOG_INFO(
168  Log() << "The node has crossed the sink threshold. Current distance=" << distance << " sink=" << m_sinkThreshold << ". Last message sent.");
169  last = true;
170  m_muteRsu.muted = true;
171  } else
172  {
173  NS_LOG_INFO(
174  Log() << "The node is still too close to the rsu. No response will be sent. Dir=" << rsu.conformantDirection << " distance=" << distance << " sink=" << m_sinkThreshold);
175  return;
176  }
177  }
178 
179  NodeInfo node;
180  node.nodeId = rsu.nodeId; //NodeId of the source
181  node.position = GetController()->GetPosition();
182  node.direction = GetController()->GetDirection();
183  node.conformantDirection = rsu.conformantDirection;
184  node.currentSpeed = GetController()->GetNodeSampler()->GetSpeed(1);
185  node.avgSpeedSmall = GetController()->GetNodeSampler()->GetSpeed(iCSInterface::AverageSpeedSampleSmall);
186  node.avgSpeedHigh = GetController()->GetNodeSampler()->GetSpeed(iCSInterface::AverageSpeedSampleHigh);
187  node.lastMessage = last;
188 
189  BeaconResponseHeader * responseHeader = new BeaconResponseHeader();
190  responseHeader->setSourceDirection(node.direction);
191  responseHeader->setCurrentSpeed(node.currentSpeed);
192  responseHeader->setAvgSpeedLow(node.avgSpeedSmall);
193  responseHeader->setAvgSpeedHigh(node.avgSpeedHigh);
194  responseHeader->setConformantDirection(node.conformantDirection.dir);
195  responseHeader->setVehicleMovement(node.conformantDirection.vMov);
196  responseHeader->setLastMessage(last);
197 
198  GetController()->SendTo(rsu.nodeId, responseHeader, PID_SPEED);
199  m_traceSendData(node);
200  NS_LOG_DEBUG(
201  Log() << "Sent beacon response to RSU " << rsu.nodeId << " for direction=" << rsu.conformantDirection << " distance=" << distance);
202  }

215  void BehaviourNodeWithoutSink::Receive(server::Payload *payload, double snr)
216  {
217  NS_LOG_FUNCTION(Log());
218  if (!m_enabled)
219  return;
220  CommHeader* commHeader;
221  GetController()->GetHeader(payload, server::PAYLOAD_FRONT, commHeader);
222  if (commHeader->getMessageType() != MT_RSU_BEACON)
223  {
224  NS_LOG_WARN(Log()<< "Received an unknown message "<< commHeader->getMessageType());
225  return;
226  }
227  BeaconHeader* beaconHeader;
228  GetController()->GetHeader(payload, server::PAYLOAD_END, beaconHeader);
229  int rsuId = commHeader->getSourceId();
230  VehicleDirection dir(beaconHeader->getDirection(), beaconHeader->getVehicleMovement());
231  NodeInfo rsu;
232  rsu.nodeId = rsuId;
233  rsu.position = commHeader->getSourcePosition();
234  rsu.conformantDirection = dir;
235  std::map<std::string, bool>::iterator it = m_activeDirections.find(dir.getId());
236  if (it == m_activeDirections.end())
237  m_activeDirections.insert(std::make_pair(dir.getId(), false));
238  if (GetController()->IsConformantDirectionAndMovement(rsu.conformantDirection, rsu.position)
239  || m_activeDirections[rsu.conformantDirection.getId()])
240  {
241 
242  double nextTime = m_rnd.GetValue(m_responseTimeSpacing,
243  beaconHeader->getMaxResponseTime() - m_responseTimeSpacing);
244  Scheduler::Cancel(m_eventResponse);
245  m_eventResponse = Scheduler::Schedule(nextTime, &BehaviourNodeWithoutSink::EventSendResponse, this, rsu);
246  NS_LOG_INFO(Log() << "scheduled a beacon response in " << nextTime << " for direction " << dir);
247  } else
248  {
249  NS_LOG_INFO(Log() << "direction "<< dir <<" not active");
250  }
251  }
