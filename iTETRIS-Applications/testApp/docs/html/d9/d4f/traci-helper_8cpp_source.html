143  bool TraciHelper::VerifyCommand(const int executionId, tcpip::Storage & sumoReply)
144  {
145  Command command;
146  if (!GetCommand(executionId, command))
147  {
148  std::ostringstream oss;
149  oss << "ExecutionId " << executionId << " not found";
150  Log::Write(oss, kLogLevelError);
151  return false;
152  }
153  int cmdLength;
154  int cmdId;
155  int resultType;
156  int cmdStart;
157  std::string msg;
158 
159  try
160  {
161  cmdStart = sumoReply.position();
162  cmdLength = sumoReply.readUnsignedByte();
163  cmdId = sumoReply.readUnsignedByte();
164  if (cmdId != command.commandId)
165  {
166  std::ostringstream oss;
167  oss << "SUMO #Error: received status response to command: " << cmdId << " but expected: "
168  << command.commandId;
169  Log::Write(oss, kLogLevelError);
170  return false;
171  }
172  resultType = sumoReply.readUnsignedByte();
173  msg = sumoReply.readString();
174  } catch (std::invalid_argument &e)
175  {
176  Log::Write("SUMO #Error: an exception was thrown while reading result state message", kLogLevelError);
177  return false;
178  }
179 
180  switch (resultType)
181  {
182  case SUMO_RTYPE_ERR:
183  {
184  std::ostringstream oss;
185  oss << "SUMO answered with error to command (" << cmdId << "), [description: " << msg << "]";
186  Log::Write(oss, kLogLevelError);
187  return false;
188  }
189  case SUMO_RTYPE_NOTIMPLEMENTED:
190  {
191  std::ostringstream oss;
192  oss << "SUMO: Sent command is not implemented (" << cmdId << "), [description: " << msg << "]";
193  Log::Write(oss, kLogLevelError);
194  return false;
195  }
196  case SUMO_RTYPE_OK:
197  {
198  std::ostringstream oss;
199  oss << "SUMO Command acknowledged (" << cmdId << "), [description: " << msg << "]";
200  Log::WriteLog(oss);
201  break;
202  }
203  default:
204  {
205  std::ostringstream oss;
206  oss << "SUMO Answered with unknown result code(" << resultType << ") to command(" << cmdId
207  << "), [description: " << msg << "]";
208  Log::WriteLog(oss);
209  return false;
210  }
211  }
212 
213  if ((cmdStart + cmdLength) != sumoReply.position())
214  {
215  std::ostringstream oss;
216  oss << "SUMO #Error: command at position " << cmdStart << " has wrong length";
217  Log::Write(oss, kLogLevelError);
218  return false;
219  }
220  return true;
221  }

223  bool TraciHelper::BeginValueRetrievalFromCommand(const int executionId, tcpip::Storage & sumoReply, int& variableId,
224  std::string & objId, int& valueType, bool alsoVerify)
225  {
226  if (alsoVerify && !VerifyCommand(executionId, sumoReply))
227  return false;
228  Command command;
229  GetCommand(executionId, command);
230  int length = sumoReply.readUnsignedByte();
231  if (length == 0)
232  length = sumoReply.readInt();
233 
234  //int respLength = inMsg.readInt();
235  int cmdId = sumoReply.readUnsignedByte();
236  if (cmdId != (command.commandId + 0x10))
237  {
238  std::ostringstream oss;
239  oss << "#Error: received response with command id: " << cmdId << " but expected: "
240  << (command.commandId + 0x10);
241  Log::Write(oss, kLogLevelError);
242  return false;
243  }
244  variableId = sumoReply.readUnsignedByte(); // variable id
245  objId = sumoReply.readString(); // object id
246  valueType = sumoReply.readUnsignedByte(); // value type
247  return true;
248  }
