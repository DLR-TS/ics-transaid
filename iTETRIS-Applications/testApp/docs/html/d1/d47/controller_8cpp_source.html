170  bool Controller::CheckDirectionAndMovement(const double &nodeDirection, const VehicleDirection &rsuDirection,
171  const double &tolerance, const Vector2D& nodePosition, const Vector2D& rsuPosition)
172  {
173  if (!CheckDirections(nodeDirection, rsuDirection.dir, tolerance))
174  return false;
175  Vector2D segmentRsuVehicle(rsuPosition.x - nodePosition.x, rsuPosition.y - nodePosition.y);
176  double segmentDir = atan2(segmentRsuVehicle.y, segmentRsuVehicle.x) * 180.0 / M_PI;
177 
178  double min = NormalizeDirection(rsuDirection.dir - 90);
179  double max = NormalizeDirection(rsuDirection.dir + 90);
180  bool result;
181  if (max > min)
182  result = segmentDir >= min && segmentDir <= max;
183  else
184  result = segmentDir >= min || segmentDir <= max;
185  if (rsuDirection.vMov == LEAVING)
186  result = !result;
187 
188  return result;
189  }
