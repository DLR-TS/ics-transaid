156  m_lanes.push_back(newLane);

185 TrafficLight* TrafficLight::CreateTrafficLight(const int index, const int numLanes,
186  const std::vector<std::string> & data, const std::string& id, const RsuData & rsuData)
187 {
188  TrafficLight* tl = new TrafficLight(id);
189  std::ostringstream oss;
190  oss << "[CreateTrafficLight] New traffic light with id " << id;
191  Log::WriteLog(oss);
192  std::vector<std::string> tokens;
193  for (int i = index; i < index + numLanes; ++i)
194  {
195  split(tokens, data[i], " ");
196  TrafficLightLane* lane = new TrafficLightLane();
197  lane->setControlledLane(tokens[1]);
198  lane->setFollowingLane(tokens[2]);
199  lane->setDirection(rsuData.lanes);
200  tl->addTrafficLightLane(lane);
201  std::ostringstream oss;
202  if (lane->getDirection() == NO_DIRECTION)
203  oss << "[CreateTrafficLight] Added new lane to the traffic light. No match found. Controlled lane: " << lane->getControlledLane() << " Following: " << lane->getFollowingLane();
204  else
205  oss << "[CreateTrafficLight] Added new lane to the traffic light. Direction: " << toString(lane->getDirection())
206  << ". Friendly name: " << lane->getFriendlyName();
207  Log::WriteLog(oss);
208  }
209  return tl;
210 }
