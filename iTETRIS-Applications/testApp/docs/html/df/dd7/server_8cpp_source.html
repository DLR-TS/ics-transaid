82  void Server::RunServer()
83  {
84  try
85  {
86  if (m_instance == NULL)
87  {
88  if (m_closeConnection)
89  return;
90  m_instance = new Server();
91  }
92 
93  while (!m_closeConnection)
94  {
95  if (!m_instance->m_inputStorage.valid_pos())
96  {
97  if (m_instance->m_outputStorage.size() > 0)
98  {
99  // send out all answers as one storage
100  m_instance->m_socket->sendExact(m_instance->m_outputStorage);
101  }
102  m_instance->m_inputStorage.reset();
103  m_instance->m_outputStorage.reset();
104  // Read a message
105  m_instance->m_socket->receiveExact(m_instance->m_inputStorage);
106  }
107  while (m_instance->m_inputStorage.valid_pos() && !m_closeConnection)
108  {
109  // dispatch each command
110  m_instance->dispatchCommand();
111  }
112  }
113  if (m_closeConnection && m_instance->m_outputStorage.size() > 0)
114  {
115  // send out all answers as one storage
116  m_instance->m_socket->sendExact(m_instance->m_outputStorage);
117  }
118  } catch (std::invalid_argument & e)
119  {
120  Console::Error("Invalid argument: ", e.what());
121 
122  } catch (SocketException &e)
123  {
124  Console::Error("Socket exception: ", e.what());
125  }
126 
127  if (m_instance != NULL)
128  {
129  delete m_instance;
130  m_instance = NULL;
131  m_closeConnection = true;
132  }
133  }

373  {
374  vector<Message> messages;
375  int numMessages = m_inputStorage.readInt();

378  Message message;

382 
383  short size = m_inputStorage.readShort();
384  ostringstream log;
385  if (size >= 14)
386  {
387 
388  //RSSI TAG
389  m_inputStorage.readUnsignedByte();
390  short rssi = m_inputStorage.readShort();
391  //TAG_SNR
392  m_inputStorage.readUnsignedByte();

394  message.m_snr = snr;
395  //TAG_TXPOWER
396  m_inputStorage.readUnsignedByte();
397  unsigned int power = m_inputStorage.readUnsignedByte();

400  int remaing = size - 14;
401  if (size >= 19)
402  {
403  //TAG_MSGID
404  m_inputStorage.readUnsignedByte();
405  int messageId = m_inputStorage.readInt();
406  log << " MSGID=" << messageId;
407  remaing = size - 19;
408  }
409  log << " [";
410  for (int tmp = 0; tmp < remaing; ++tmp)
411  {
412  log << (int) m_inputStorage.readChar() << ",";
413  }
414  log << "]";
415 
416  } else
417  {
418  log << "[" << message.m_destinationId << "|" << message.m_messageId << "|" << message.m_extra << "] ";
419  log << "size= " << size << " [";
420  for (int tmp = 0; tmp < size; tmp++)
421  {
422  log << (int) m_inputStorage.readChar() << ",";
423  }
424  log << "]";
425  }

427  Log::WriteLog(log);
428  }
429  m_nodeHandler->applicationMessageReceive(messages);
430  writeStatusCmd(CMD_APP_MSG_RECEIVE, APP_RTYPE_OK, "CMD_APP_MSG_RECEIVE");

445  bool Server::applicationExecute()
446  {
447  int nodeId = m_inputStorage.readInt();
448  //Reset the last seen counter
449  std::map<int, int>::iterator it = m_lastSeenNodes.find(nodeId);
450  if (it == m_lastSeenNodes.end())
451  m_lastSeenNodes.insert(std::make_pair(nodeId, 0));
452  else
453  it->second = 0;
454  // create reply message
455  writeStatusCmd(CMD_NOTIFY_APP_EXECUTE, APP_RTYPE_OK, "CMD_NOTIFY_APP_EXECUTE");
456  DirectionValueMap data;
457  if (m_nodeHandler->applicationExecute(nodeId, data))
458  {
459  Storage dataStorage;
460  dataStorage.writeUnsignedByte(data.size());
461  for (DirectionValueMap::const_iterator dirIt = data.begin(); dirIt != data.end(); ++dirIt)
462  {
463  dataStorage.writeString(dirIt->first);
464  dataStorage.writeUnsignedByte(dirIt->second.size());
465  for (ValueMap::const_iterator valueIt = dirIt->second.begin(); valueIt != dirIt->second.end(); ++valueIt)
466  {
467  dataStorage.writeUnsignedByte(valueIt->first);
468  dataStorage.writeDouble(valueIt->second);
469  }
470  }
471  //data
472  Storage tmpStorage;
473  tmpStorage.writeInt(1); //Num of result
474  tmpStorage.writeUnsignedByte(TRAFFIC_INFO); //Command type
475  tmpStorage.writeString("TrafficData"); //Tag
476  tmpStorage.writeInt(dataStorage.size()); //size of data
477  tmpStorage.writeStorage(dataStorage); //data
478 
479  m_outputStorage.writeInt(4 + 1 + 1 + tmpStorage.size());
480  m_outputStorage.writeUnsignedByte(CMD_NOTIFY_APP_EXECUTE);
481  m_outputStorage.writeUnsignedByte(APP_RESULT_ON);
482  m_outputStorage.writeStorage(tmpStorage);
483  } else
484  {
485  m_outputStorage.writeInt(4 + 1 + 1);
486  m_outputStorage.writeUnsignedByte(CMD_NOTIFY_APP_EXECUTE);
487  m_outputStorage.writeUnsignedByte(APP_RESULT_OFF);
488  }
489  return true;
490  }
