288  void OutputHelper::OnPacketSend(iCSInterface * controller, server::Payload * payload)
289  {
290  if (m_samplePackets)
291  {
292  // check if sender is inside a sector
293  if (IsVehicle(controller->GetNodeType()))
294  {
295  // compute node distance
296  Vector2D pos = controller->GetPosition();
297  const Vector2D rsuPosition = m_rsuController->GetPosition();
298  double distance = GetDistance(rsuPosition, pos);
299  if (distance <= mp_sink_threshold_max)
300  {
301  // retrieve node direction by its position (reverted)
302  double dx = rsuPosition.x - pos.x;
303  double dy = rsuPosition.y - pos.y;
304  double ang = atan2(dy, dx) * 180 / M_PI;
305  for (std::vector<Sector>::iterator it = m_sectors.begin(); it != m_sectors.end(); ++it)
306  {
307  if (iCSInterface::CheckDirectionAndMovement(ang, it->direction, mp_sink_tolerance, pos, rsuPosition))
308  {
309  it->packetCount++;
310  break;
311  }
312  }
313  }
314  }
315  }
316  m_psent++;
317  }

396  void OutputHelper::GMSample()
397  {
398  using namespace server;
399  Node * node;
400  NodeMeta* meta;
401 
402  // check every node
403  unsigned int nodeCount = 0;
404  unsigned int nodeCountMobSampling = 0;
405  std::vector<std::string> nList;
406  NodeHandler* handler = Server::GetNodeHandler();
407  const Vector2D rsuPosition = m_rsuController->GetPosition();
408  for (NodeMap::const_iterator it = handler->begin(); it != handler->end(); ++it)
409  //for (std::map<int, OutputHelperWrapper *>::const_iterator it = m_wrappers.begin(); it != m_wrappers.end(); ++it)
410  {
411  node = it->second;
412 
413  if (IsNodeType(node->getNodeType(), NT_RSU))
414  continue; // do not sample rsu nodes
415 
416  // prepare node metadata
417  meta = GetMeta(node, true);
418  meta->ResetSample();
419 
420  // compute node distance
421  Vector2D pos = node->getPosition();
422  double distance = GetDistance(rsuPosition, pos);
423 
424  if (distance <= mp_sink_threshold_max)
425  {
426  // node is inside sink range
427  nodeCount++;
428  meta->flagAsEnteredRsuRange = true;
429  meta->sample_enteredRsuRange = true;
430 
431  if (distance > mp_sink_threshold_min)
432  {
433  //get direction from sumo
434  double ang = node->getDirection();
435  // retrieve node direction by its position (reverted)
436  /*double dx = rsuPosition.x - pos.x;
437  double dy = rsuPosition.y - pos.y;
438  double ang = atan2(dy, dx) * 180 / M_PI;*/
439  /*
440  Vector2D dir = VectorToVector2D (mobility->GetVelocity ());
441  double ang = atan2 (dir.y, dir.x) * 180 / M_PI;
442  */
443  bool found = false;
444  for (std::vector<Sector>::iterator sIt = m_sectors.begin(); sIt != m_sectors.end(); ++sIt)
445  {
446  Vector2D segmentRsuVehicle(rsuPosition.x - pos.x, rsuPosition.y - pos.y);
447  double segmentDir = atan2(segmentRsuVehicle.y, segmentRsuVehicle.x) * 180.0 / M_PI;
448 // NS_LOG_DEBUG(
449 // "[GMSample][" << node->getId() << "] dir=" << ang << " sd=" << segmentDir << " rp=" << rsuPosition << " np=" << pos);
450  if (iCSInterface::CheckDirectionAndMovement(ang, sIt->direction, mp_sink_tolerance, pos, rsuPosition))
451  {
452  found = true;
453  // node is inside sector
454  if (!meta->flagAsEnteredSector)
455  {
456  // first time this node enters a sector
457  meta->flagAsEnteredSector = true;
458  meta->enteredRsuDirection = sIt->direction;
459  }
460 // else if (meta->enteredRsuDirection != it->direction)
461 // break; // node comes from another sector
462 
463  meta->sample_enteredRsuDirection = sIt->direction;
464  sIt->nodeCount++;
465  NS_LOG_DEBUG("[GMSample][" << node->getId() << "] dir=" << sIt->direction);
466  sIt->avgSpeed += GetRelativeSpeed(node->getVelocity(), m_rsuController->GetNode()->getVelocity());
467  break;
468  }
469  }
470  if (!found)
471  NS_LOG_DEBUG("[GMSample][" << node->getId() << "] dir=" << ang << " not found.");
472  }
473  }
474  }
475  // print sample statistics
476  std::ostringstream strs;
477  strs << "real totc=" << nodeCount << " : ";
478  for (std::vector<Sector>::const_iterator it = m_sectors.begin(); it != m_sectors.end(); ++it)
479  {
480  // for each sector:
481  // log node count
482  strs << it->direction.getId() << "=" << it->nodeCount << ";";
483  // log avg speed
484  if (it->nodeCount == 0)
485  strs << 0;
486  else
487  strs << (it->avgSpeed / it->nodeCount);
488  // log packet sent
489  strs << ";" << it->packetCount << " ";
490  }
491  Log(m_rsuController, strs.str());
492  SectorReset();
493 
494  // accumulate results
495  m_rsuDensityAccum += nodeCount;
496  m_rsuDensityCount++;
497  if (nodeCount > m_maxRsuDensity)
498  m_maxRsuDensity = nodeCount;
499 
500  // schedule next sample
501  Scheduler::Schedule(mp_t_sample, &OutputHelper::GMSample, this);
502 
503  }
