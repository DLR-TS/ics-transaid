100 int PayloadStorage::expiredPayloadCleanUp(const int oldTimeStep)
101 {
102  int number = 0;
103  ostringstream log;
104  log << "ExpiredPayloadCleanUp for time: " << oldTimeStep << ". Removing keys: ";
105  for (map<unsigned, const Payload*>::iterator it = m_storage.begin(); it != m_storage.end();)
106  {
107  if (it->second->getTimeStep() <= oldTimeStep)
108  {
109  log << it->first << " ";
110  ++number;
111  delete (it->second);
112  m_storage.erase(it++);
113  }
114  else
115  ++it;
116  }
117  if (number == 0)
118  log << "noting to remove ";
119  log << "Current size: " << m_storage.size();
120  Log::WriteLog(log);
121  return number;
122 }
