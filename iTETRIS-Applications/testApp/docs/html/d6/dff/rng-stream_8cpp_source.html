69 // Return (a*s + c) MOD m; a, s, c and m must be < 2^35
70 //
71 double MultModM (double a, double s, double c, double m)
72 {
73  double v;
74  int32_t a1;
75 
76  v = a * s + c;
77 
78  if (v >= two53 || v <= -two53)
79  {
80  a1 = static_cast<int32_t> (a / two17);
81  a -= a1 * two17;
82  v = a1 * s;
83  a1 = static_cast<int32_t> (v / m);
84  v -= a1 * m;
85  v = v * two17 + a * s + c;
86  }
87 
88  a1 = static_cast<int32_t> (v / m);
89  /* in case v < 0)*/
90  if ((v -= a1 * m) < 0.0)
91  {
92  return v += m;
93  }
94  else
95  {
96  return v;
97  }
98 }

105 void MatVecModM (const Matrix A, const double s[3], double v[3],

217 // The following are the transition matrices of the two MRG components
218 // (in matrix form), raised to all powers of 2 from 1 to 191
219 struct Precalculated

222  Matrix a2[190];

224 struct Precalculated PowerOfTwoConstants (void)
225 {
226  struct Precalculated precalculated;

253 // Generate the next random number.
254 //
255 double RngStream::RandU01 ()
256 {
257  int32_t k;
258  double p1, p2, u;
259 
260  /* Component 1 */
261  p1 = a12 * m_currentState[1] - a13n * m_currentState[0];
262  k = static_cast<int32_t> (p1 / m1);
263  p1 -= k * m1;
264  if (p1 < 0.0)
265  {
266  p1 += m1;
267  }
268  m_currentState[0] = m_currentState[1]; m_currentState[1] = m_currentState[2]; m_currentState[2] = p1;
269 
270  /* Component 2 */
271  p2 = a21 * m_currentState[5] - a23n * m_currentState[3];
272  k = static_cast<int32_t> (p2 / m2);
273  p2 -= k * m2;
274  if (p2 < 0.0)
275  {
276  p2 += m2;
277  }
278  m_currentState[3] = m_currentState[4]; m_currentState[4] = m_currentState[5]; m_currentState[5] = p2;
279 
280  /* Combination */
281  u = ((p1 > p2) ? (p1 - p2) * norm : (p1 - p2 + m1) * norm);
282 
283  return u;
284 }
