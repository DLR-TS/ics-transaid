153  bool Node::askForSubscription(const int subscriptionId, tcpip::Storage * & request)
154  {
155  if (m_type == NT_VEHICLE_SHADOW)
156  return false;
157  if (m_firstAskSubscription)
158  {
159  m_firstAskSubscription = false;
160  addSubscriptions();
161  } else
162  {
163  // remove the subscription I used the last time. I delete it now otherwise I can't delete the request storage
164  delete m_toSubscribe.front();
165  m_toSubscribe.pop();
166  }
167  if (m_toSubscribe.empty())
168  {
169  m_firstAskSubscription = true;
170  return false;
171  }
172  SubscriptionHolder* subscription = m_toSubscribe.front();
173  request = subscription->m_request;
174  addSubscription(subscriptionId, subscription->m_subscriptionType, subscription->m_toUnsubscribe);
175  return true;
176  }
