138  SubscriptionHolder * SubscriptionHelper::ReceiveUnicast(const int destinationId, const std::vector<int> & sourceIds)
139  {
140  Storage* request = new Storage();
141 
142  int variableSize = 0;
143  if (sourceIds.size() > 0)
144  variableSize = 2 + 4 * sourceIds.size();
145  request->writeUnsignedByte(1 + 1 + 1 + 1 + 1 + 4 + 2 + 1 + 1 + variableSize);
146  request->writeUnsignedByte(CMD_ASK_FOR_SUBSCRIPTION);
147  request->writeUnsignedByte(SUB_APP_MSG_RECEIVE);
148  // Message type, not used by iCS for unicast
149  request->writeUnsignedByte(0x01);
150  // Only destID
151  request->writeUnsignedByte(0x04);
152  //destination of the unicast message
153  request->writeInt(destinationId);
154  //payload length.. useless
155  request->writeShort(0);
156  request->writeUnsignedByte(EXT_HEADER_TYPE_UNICAST);
157  if (sourceIds.size() > 0)
158  {
159  request->writeUnsignedByte(EXT_HEADER__VALUE_BLOCK_IDs);
160  request->writeShort(sourceIds.size());
161  for (std::vector<int>::const_iterator it = sourceIds.begin(); it != sourceIds.end(); ++it)
162  request->writeInt(*it);
163  } else
164  {
165  //use a different value to select any source
166  request->writeUnsignedByte(EXT_HEADER__VALUE_BLOCK_IDs + 1);
167  }
168 
169  return new SubscriptionHolder(SUB_APP_MSG_RECEIVE, request);
170  }

172  SubscriptionHolder * SubscriptionHelper::ReceiveGeobroadcast(const int messageId, const Circle * propagationCircle)
173  {
174  Storage* request = new Storage();
175  int variableSize = 0;
176  if (propagationCircle != NULL)
177  variableSize = 1 + 1 + 4 + 4 + 4;
178  request->writeUnsignedByte(1 + 1 + 1 + 1 + 1 + 2 + 1 + 1 + variableSize);
179  request->writeUnsignedByte(CMD_ASK_FOR_SUBSCRIPTION);
180  request->writeUnsignedByte(SUB_APP_MSG_RECEIVE);
181  // Message type, used as message Id
182  request->writeUnsignedByte(messageId);
183  // Do not set anything
184  request->writeUnsignedByte(0x00);
185  //payload length.. useless
186  request->writeShort(0);
187  request->writeUnsignedByte(EXT_HEADER_TYPE_GEOBROADCAST);
188  if (propagationCircle != NULL)
189  {
190  request->writeUnsignedByte(EXT_HEADER__VALUE_BLOCK_AREAs);
191  //number areas. Use only one
192  request->writeUnsignedByte(0x01);
193  request->writeUnsignedByte(EXT_HEADER__CIRCLE);
194  request->writeFloat(propagationCircle->x);
195  request->writeFloat(propagationCircle->y);
196  request->writeFloat(propagationCircle->radius);
197  } else
198  {
199  //use a different value to select any area
200  request->writeUnsignedByte(EXT_HEADER__VALUE_BLOCK_AREAs + 1);
201  }
202  delete propagationCircle;
203 
204  return new SubscriptionHolder(SUB_APP_MSG_RECEIVE, request);
205  }

213  SubscriptionHolder * SubscriptionHelper::SendUnicast(const int sourceId, const short payloadLength,
214  const int messageId, const std::vector<int> & destinationIds, const std::string & extra, const double time)
215  {
216  Storage* request = new Storage();
217  int payloadSize = 1 + 4 + extra.size();
218  int variableSize = 0;
219  if (destinationIds.size() > 0)
220  variableSize = (4 + 2 + payloadSize) * destinationIds.size();
221  request->writeUnsignedByte(1 + 1 + 1 + 1 + 1 + 1 + 1 + 4 + 1 + 8 + 2 + 4 + 1 + 1 + 1 + variableSize);
222  request->writeUnsignedByte(CMD_ASK_FOR_SUBSCRIPTION);
223  request->writeUnsignedByte(SUB_APP_MSG_SEND);
224  // Message type, not used in a send
225  request->writeUnsignedByte(33);
226  // in bits, it is: 11111 : the preferred techno, comm profile, senderID, the message lifetime and the precise timestep
227  request->writeUnsignedByte(0x1F);
228  request->writeUnsignedByte(0xFF); //preferred techno
229  request->writeUnsignedByte(0x00); //Comm profile. Set to 0 or it will not send messages of mobile nodes
230  request->writeInt(sourceId); //senderID
231  request->writeUnsignedByte(2); //message lifetime
232  request->writeDouble(time); //precise timestep
233  request->writeShort(payloadLength);
234  // Message sequence number. used as message id
235  request->writeInt(messageId);
236  request->writeUnsignedByte(EXT_HEADER_TYPE_UNICAST); // CommMode
237  request->writeUnsignedByte(EXT_HEADER__VALUE_BLOCK_IDs); // CommMode
238  request->writeUnsignedByte(destinationIds.size());
239  if (destinationIds.size() > 0)
240  {
241  for (std::vector<int>::const_iterator it = destinationIds.begin(); it != destinationIds.end(); ++it)
242  {
243  request->writeInt(*it);
244  //payload size
245  request->writeShort(payloadSize);
246  request->writeUnsignedByte(EXT_HEADER__EXTRA);
247  request->writeString(extra);
248  }
249  }
250 
251  return new SubscriptionHolder(SUB_APP_MSG_SEND, request);
252  }
