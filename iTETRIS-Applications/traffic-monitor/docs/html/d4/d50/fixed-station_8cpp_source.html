121  void FixedStation::trafficLightInformation(const bool error, const std::vector<std::string> & data)
122  {
123  std::vector<std::string> tokens;
124  if (error)
125  {
126  split(tokens, data[0], ":");
127  std::ostringstream oss;
128  oss << "[TrafficLightInformation] error: " << tokens[1] << ". Will unsubscribe " << tokens[0];
129  Log::Write(oss, LOG_ERROR);
130  int subNo = stringToInt(tokens[0]);
131  setToUnsubscribe(subNo);
132  } else
133  {
134  for (int i = 0; i < data.size(); ++i)
135  {
136  split(tokens, data[i], " ");
137  if (tokens.size() > 2)
138  {
139  int numLanes = stringToInt(tokens[2]);
140  m_trafficLight = TrafficLight::CreateTrafficLight(i + 1, numLanes, data, tokens[0], m_rsuData);
141  i += numLanes;
142  }
143  m_trafficLight->setState(tokens[1]);
144  break; //Will use only the first traffic light from the subscription
145  }
146  std::ostringstream oss;
147  oss << "[TrafficLightInformation] Traffic light status update: " << m_trafficLight->getState();
148  Log::WriteLog(oss);
149  }
150  }
