69  void Scheduler::Cancel(event_id & id)
70  {
71  if (id == 0)
72  return;
73  for (multiset<Event*>::iterator it = m_list.begin(); it != m_list.end(); ++it)
74  {
75  if ((*it)->m_id == id)
76  {
77  //Can't cancel the current invocation here. Will be removed at the end of the invocation by the Notify
78  if (id != m_currentInvoke)
79  {
80  delete *it;
81  m_list.erase(it);
82  }
83  //Set it to zero so subsequent calls will avoid entering the loop
84  id = 0;
85  return;
86  }
87  }

90  int Scheduler::Notify(int currentTime)
91  {
92  NS_LOG_INFO("[Scheduler] Notify time: " << currentTime << ". Current size: " << m_list.size());
93  int number = 0;
94  for (multiset<Event*>::iterator it = m_list.begin(); it != m_list.end();)
95  {
96  Event* e = *it;
97  if (currentTime >= e->m_time)
98  {
99  //I need to save the id of the current invocation so I don't remove it in Cancel
100  m_currentInvoke = e->m_id;
101  m_currentTime = e->m_time;
102  e->m_callBack->invoke();
103  ++number;
104  delete e;
105  m_list.erase(it++);
106  } else
107  {
108  //The elements are ordered by time so I can exit the loop the first time the condition is not satisfied
109  break;
110  }
111  }
112  //Zero is invalid as Id
113  m_currentInvoke = 0;
114  m_currentTime = currentTime;
115  NS_LOG_INFO("[Scheduler] " << number << " have been executed.");
116  if (m_list.size() > 0)
117  {
118  Event* begin = *(m_list.begin());
119  NS_LOG_INFO(
120  "[" << currentTime << "][Scheduler] next at " << begin->m_time << " in " << (begin->m_time - currentTime));
121  }
122  return number;
123  }
