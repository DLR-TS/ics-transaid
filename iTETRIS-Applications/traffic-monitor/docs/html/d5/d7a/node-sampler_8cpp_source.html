130  Vector2D NodeSampler::GetSpeedVector(int numSamples) const
131  {
132  if (m_speedBuffer->size() == 0 || numSamples < 1)
133  return Vector2D();
134  if (m_speedBuffer->size() < numSamples)
135  numSamples = m_speedBuffer->size();
136  // compute vector mean
137  double x = 0, y = 0;
138  for (unsigned int i = 0; i < numSamples; i++)
139  {
140  x += m_speedBuffer->at(i).x;
141  y += m_speedBuffer->at(i).y;
142  }
143  x /= (double) numSamples;
144  y /= (double) numSamples;
145 
146  return Vector2D(x, y);

248  double NodeSampler::SteerFilter(double sample)
249  {
250  static const double RAD2DEG = 180.0f / M_PI;
251  static const double DEG2RAD = M_PI / 180.0f;
252  if (m_lastDirection == DIR_INVALID || sample == DIR_INVALID)
253  return sample;
254 
255  double sampleDEG = sample * RAD2DEG;
256  double diff = AngleDifference(m_lastDirection, sampleDEG);
257  if (std::abs(diff) > m_steerFilterThreshold)
258  {
259  // sample is over threshold
260  if (m_currentSteerFilterCount > m_steerFilterCount)
261  {
262  m_currentSteerFilterCount = 0;
263  m_directionBuffer->clear();
264  return sample;
265  } else
266  {
267  m_currentSteerFilterCount++;
268  return m_lastDirection * DEG2RAD;
269  }
270  } else if (m_currentSteerFilterCount)
271  {
272  m_currentSteerFilterCount--;
273  }
274  return sample;
275  }

294  void NodeSampler::Sample()
295  {
296  NS_LOG_FUNCTION(m_controller->NodeName() << ": NodeSampler: ");
297  // schedule next sampling
298  m_eventSample = Scheduler::Schedule(m_resolution, &NodeSampler::Sample, this);
299 
300  if (m_currentVehicle == NULL)
301  {
302  NS_LOG_WARN(m_controller->NodeName() << ": Current vehicle is NULL");
303  return;
304  }
305  // sample current speed & position
306  Vector2D position = m_currentVehicle->position;
307  //MYCOM could modify the function to avoid this conversion
308  //Add error to speed value, only if not 0
309  double spd = m_currentVehicle->speed;
310  if (spd != 0)
311  spd += m_speedError * m_speedErrorVariable.GetValue();
312  //Get the speed components from module and angle
313  double speed_x = (double) spd * cos((double) m_currentVehicle->direction * M_PI / 180.0);
314  double speed_y = (double) spd * sin((double) m_currentVehicle->direction * M_PI / 180.0);
315  Vector2D speed(speed_x, speed_y);
316 
317  // store speed sample. Need to be stored here otherwise the speed will not go to zero when the vehicle has stopped
318  m_speedBuffer->push_front(speed);
319 
320  // sample direction
321  double direction;
322  if (std::abs(speed.x) < 0.01 && std::abs(speed.y) < 0.01)
323  {
324  // movement speed is too low: direction is unreliable
325  direction = DIR_INVALID;
326  // if present, we recycle the previous direction sample
327  if (m_directionBuffer->size() > 0)
328  direction = m_directionBuffer->front();
329  } else
330  direction = atan2(speed.y, speed.x);
331 
332  // compute real distance between previous sample
333  double dist = std::sqrt(pow(m_lastPosition.x - position.x, 2) + pow(m_lastPosition.y - position.y, 2));
334  if (dist < m_movimentThreshold)
335  {
336  // the node didn't move enough:
337  // discard this sample
338  NS_LOG_DEBUG(
339  m_controller->NodeName() << ": sample dropped. The node didn't move enough: " << dist << " . Min is: " << m_movimentThreshold);
340  return;
341  }
342 
343  // compute position error
344  double radius = m_positionErrorRadius.GetValue();
345  double angle = m_positionErrorAngle.GetValue();
346  Vector2D position_err = Vector2D(position.x + radius * std::cos(angle), position.y + radius * std::sin(angle));
347  // compute direction error
348  double direction_err = DIR_INVALID;
349  if (direction != DIR_INVALID)
350  direction_err = direction + m_directionErrorAngle.GetValue();
351 
352  // apply steer filter
353  direction_err = SteerFilter(direction_err);
354 
355  // store position & direction samples
356  m_positionBuffer->push_front(position_err);
357  m_directionBuffer->push_front(direction_err);
358  m_lastPosition = position;
359 
360  // update means
361  m_direction = ComputeDirection();
362  m_traceDirection(m_direction);
363  m_position = ComputePosition();
364  m_tracePosition(m_position);
365  m_lastDirection = m_direction;
366 
367  NS_LOG_INFO(
368  m_controller->NodeName() << ": pos=(" << position_err.x << ", " << position_err.y << ") spd=(" << speed.x << ", " << speed.y << ") avgDir=" << formatInvalid(m_direction, false) << " dir=" << formatInvalid(direction_err));
369  NS_LOG_DEBUG(
370  m_controller->NodeName() << ": real: dir= " << formatInvalid(direction) << " pos=(" << position.x << ", " << position.y << ") dist= " << dist);
371  }
