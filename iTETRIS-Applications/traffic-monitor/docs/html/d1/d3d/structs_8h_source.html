124  MobilityInfo(tcpip::Storage & storage)
125  {
126  id = storage.readInt();
127  double x = storage.readFloat();
128  double y = storage.readFloat();
129  //Don not read inline x and y because it will read y before...
130  position = application::Vector2D(x, y);
131  isMobile = storage.readUnsignedByte() == 0 ? false : true;
132  if (isMobile)
133  {
134  speed = storage.readFloat();
135  direction = ConvertDirection(storage.readFloat());
136  acceleration = storage.readFloat();
137  lane = storage.readString();
138  }
139  }
